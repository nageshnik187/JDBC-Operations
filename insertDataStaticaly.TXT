package JDBC;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.Statement;
import java.io.*;
public class JDBCInsertData {

	public static void main(String[] args) {

try{
			
			Class.forName("com.mysql.cj.jdbc.Driver");
			Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/youtube", "root", "12345");
			
            String q = "INSERT INTO Table1(FirstName, LastName, City) values(?,?,?)";
            	     
             PreparedStatement stmt= con.prepareStatement(q);
             
            
            stmt.setString(1, "Nagesh");
            stmt.setString(2, "Nikavade");
            stmt.setString(3, "pune");
             
             
			stmt.executeUpdate();
			System.out.println("inserted....");
			
			con.close();
		}
		catch(Exception e){
			e.printStackTrace();
		}

	}

}
